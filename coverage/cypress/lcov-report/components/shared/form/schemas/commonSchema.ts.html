import { z } from 'zod'
 
/**
 * 共通のスキーマ定義
 */
 
// 基本的な入力フィールドのバリデーション
export const requiredString = z.string().min(1, '必須項目です')
export const optionalString = z.string().optional()
export const email = z.string().email('有効なメールアドレスを入力してください').optional().or(z.literal(''))
export const phone = z.string().regex(/^[0-9-]*$/, '数字とハイフンのみ使用できます').optional()
 
// 金額関連のバリデーション
export const amount = z.number().min(0, '0以上の数値を入力してください')
export const positiveAmount = z.number().positive('0より大きい数値を入力してください')
export const taxRate = z.number()
  .min(0, '0以上の数値を入力してください')
  .max(1, '1以下の数値を入力してください')
  .transform(value => Number(value.toFixed(2)))
 
// 日付関連のバリデーション
export const requiredDate = z.date({
  required_error: '日付を入力してください',
  invalid_type_error: '有効な日付を入力してください',
})
export const optionalDate = z.date().optional()
 
// タグのスキーマ
export const tagSchema = z.object({
  id: z.string().optional(),
  name: z.string().min(1, 'タグ名は必須です')
})
 
export type Tag = z.infer<typeof tagSchema>
 
// 銀行情報のスキーマ
export const bankInfoSchema = z.object({
  bankName: z.string().min(1, '銀行名は必須です'),
  branchName: z.string().min(1, '支店名は必須です'),
  accountType: z.enum(['ORDINARY', 'CURRENT', 'SAVINGS']),
  accountNumber: z.string().min(1, '口座番号は必須です'),
  accountHolder: z.string().min(1, '口座名義は必須です')
})
 
export type BankInfo = z.infer<typeof bankInfoSchema>
 
// 明細行のスキーマ
export const itemSchema = z.object({
  id: z.string().optional(),
  itemName: z.string().min(1, '品目名は必須です'),
  quantity: z.number()
    .positive('数量は0より大きい値を入力してください')
    .min(1, '数量は1以上を入力してください'),
  unitPrice: z.number()
    .min(0, '単価は0以上を入力してください'),
  taxRate: z.number()
    .min(0.1, 'インボイス制度に基づき、税率は10%以上を入力してください')
    .max(1, '税率は100%以下を入力してください')
    .transform(value => Number(value.toFixed(2))),
  description: z.string().optional()
})
 
export type Item = z.infer<typeof itemSchema>
